    Create `src/presentation/hooks/useSeatPlan.ts`:
    ```ts
    import { useQuery } from '@tanstack/react-query';
    import { services } from '@/infrastructure/DependencyProvider';

    export function useSeatPlan(flightId: string) {
      return useQuery({
        queryKey: ['seatPlan', flightId],
        queryFn: () => services.seatQuery.getSeatPlan(flightId),
        enabled: Boolean(flightId),
        staleTime: 5 * 60 * 1000,
      });
    }
    ```

    For tests: wrap in a custom QueryClientProvider with `gcTime: 0` to prevent stale cache between tests.
    Mock `services.seatQuery` via `vi.mock('@/infrastructure/DependencyProvider')`.
    Use `renderHook` from `@testing-library/react` with a wrapper providing fresh QueryClient per test.

    Test setup pattern:
    ```tsx
    const wrapper = ({ children }) => (
      <QueryClientProvider client={new QueryClient({ defaultOptions: { queries: { retry: false } } })}>
        {children}
      </QueryClientProvider>
    );
    const { result } = renderHook(() => useSeatPlan('FL001'), { wrapper });
    await waitFor(() => expect(result.current.isSuccess).toBe(true));
    ```
