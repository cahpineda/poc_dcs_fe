```tsx
import type { Seat } from '../../domain/seat/Seat';

export interface PassengerDetailDrawerProps {
  seat: Seat | null;
  onClose: () => void;
  onReseat: (seat: Seat) => void;
  onBlock: (seat: Seat) => void;
  onUnblock: (seat: Seat) => void;
}

const STATUS_LABEL: Record<string, string> = {
  occupied: 'Occupied',
  checked_in: 'Checked-in',
  boarded: 'Boarded',
  blocked: 'Blocked',
};

const BLOCKABLE = new Set(['occupied', 'checked_in', 'boarded']);

export function PassengerDetailDrawer({ seat, onClose, onReseat, onBlock, onUnblock }: PassengerDetailDrawerProps) {
  if (!seat) return null;

  const statusLabel = STATUS_LABEL[seat.status] ?? seat.status;
  const showBlock = BLOCKABLE.has(seat.status);
  const showUnblock = seat.status === 'blocked';
  const showReseat = BLOCKABLE.has(seat.status); // reseat only meaningful for occupied seats

  return (
    <aside className="passenger_drawer" role="complementary" aria-label="Passenger details">
      <header className="passenger_drawer__header">
        <h3 className="passenger_drawer__title">{seat.passengerName ?? '—'}</h3>
        <button
          type="button"
          className="passenger_drawer__close"
          aria-label="Close drawer"
          onClick={onClose}
        >
          ×
        </button>
      </header>
      <dl className="passenger_drawer__meta">
        <dt>Seat</dt>
        <dd>{String(seat.number)}</dd>
        <dt>Status</dt>
        <dd>{statusLabel}</dd>
      </dl>
      <div className="passenger_drawer__actions">
        {showReseat && (
          <button type="button" onClick={() => onReseat(seat)}>Reseat</button>
        )}
        {showBlock && (
          <button type="button" onClick={() => onBlock(seat)}>Block seat</button>
        )}
        {showUnblock && (
          <button type="button" onClick={() => onUnblock(seat)}>Unblock seat</button>
        )}
      </div>
    </aside>
  );
}
```

WHY each detail:
- `if (!seat) return null` makes the "renders nothing when null" test pass cleanly.
- `BLOCKABLE` Set drives showBlock/showReseat — keeps the rules in one constant.
- `STATUS_LABEL` map keeps status-to-label mapping declarative; tests assert the exact strings.
- `aria-label="Close drawer"` is the test-discoverable handle for the close button (do NOT change the string — tests rely on it).

Avoid: adding pending/loading state in this task (that's plan 10-02), adding CSS-in-JS, adding animation logic (CSS handles slide-in).
