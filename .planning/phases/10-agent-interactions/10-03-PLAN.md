<objective>
Implement the reseat workflow: agent opens drawer on an occupied seat, clicks Reseat, the seat map enters "reseat mode" (only available/exit_row_available seats clickable, banner visible), agent picks new seat, ReseatPassengerCommand fires, on success the cache invalidates and reseat mode exits.

Purpose: This is the core agent workflow that motivates the entire phase. Cloud_2 supports it as a multi-step interaction, and the POC must match. It also requires propagating a `reseatMode` prop through the component tree (SeatMap → CabinDeck → SeatRow → SeatCell) so SeatCell can narrow its clickable-status set.

Output:
- New `useSeatReseat(flightId)` hook with test coverage
- Three new pieces of state in SeatPlanTab: `reseatPassengerId`, `reseatFromSeat`, `reseatPassengerName`
- `reseatMode` prop threaded through SeatMap → CabinDeck → SeatRow → SeatCell
- SeatCell narrows `CLICKABLE_STATUSES` to `available` + `exit_row_available` when `reseatMode=true`
- Reseat banner with passenger name + Cancel button, styled in feedback.css
- Tests covering: hook contract, prop threading, status narrowing, command dispatch, success cleanup, cancel flow
</objective>

```ts
import { useMutation, useQueryClient } from '@tanstack/react-query';
import type { ReseatPassengerCommand } from '../../application/commands/ReseatPassengerCommand';
import { services } from '../../infrastructure/services'; // adjust import to match useSeatAssign

export function useSeatReseat(flightId: string) {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (command: ReseatPassengerCommand) =>
      services.seatCommand.reseatPassenger(command),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['seatPlan', flightId] });
    },
  });
}
```

```ts
function handleSeatSelect(seat: Seat) {
  // Reseat mode takes precedence
  if (reseatMode && reseatPassengerId && reseatFromSeat) {
    if (seat.status !== 'available' && seat.status !== 'exit_row_available') {
      return; // SeatCell already gates clicks, but defense in depth

        fromSeat: reseatFromSeat,
        toSeat: String(seat.number),
        flightId,

          setReseatFromSeat(null);
          setReseatPassengerName(null);
        },
      }
    );
    return;

  const isOccupiedLike = seat.status === 'occupied'
    || seat.status === 'checked_in'
    || seat.status === 'boarded'
    || seat.status === 'blocked';
```
