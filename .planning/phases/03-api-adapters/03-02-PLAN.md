    ```ts
    export type SeatStatus =
      | 'available'
      | 'occupied'
      | 'blocked'
      | 'exit_row_available'
      | 'exit_row_occupied'
      | 'unavailable'
      | 'infant_occupied';
    ```

    - Private constructor storing SeatNumber, SeatStatus, cabinClass, isExitRowSeat, hasInfant

    - isAvailable(): boolean — status is 'available' or 'exit_row_available'
    - isExitRow(): boolean — status starts with 'exit_row_'
    - isOccupied(): boolean — status ends with '_occupied' or is 'occupied'
    - withStatus(status: SeatStatus): Seat — returns new Seat with updated status (immutable)
    - get number(): SeatNumber
    - get status(): SeatStatus
    - get cabin(): string
