    ```ts
    import type { AxiosInstance } from 'axios';
    import type { ISeatCommandService } from '@/application/ports';

    export class Cloud2SeatCommandAdapter implements ISeatCommandService {
      constructor(private readonly http: AxiosInstance) {}

      async assignSeat(command: unknown): Promise<void> {
        if (!command) throw new Error('assignSeat: command is required');
        await this.http.post('/ajax/seat_plan/assign_seat', command);
      }

      async reseatPassenger(command: unknown): Promise<void> {
        if (!command) throw new Error('reseatPassenger: command is required');
        await this.http.post('/ajax/seat_plan/reseat_passenger', command);
      }

      async blockSeat(command: unknown): Promise<void> {
        if (!command) throw new Error('blockSeat: command is required');
        await this.http.post('/ajax/seat_plan/block_seat', command);
      }

      async unblockSeat(command: unknown): Promise<void> {
        if (!command) throw new Error('unblockSeat: command is required');
        await this.http.post('/ajax/seat_plan/unblock_seat', command);
      }
    }
    ```

    Handle HTTP 409 in assignSeat: catch AxiosError with status 409, extract seat number from command, throw domain Error.
