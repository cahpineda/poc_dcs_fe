    Create `src/infrastructure/DependencyProvider.tsx`:
    ```tsx
    import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
    import { ReactQueryDevtools } from '@tanstack/react-query-devtools';
    import type { ReactNode } from 'react';
    import type { ISeatPlanQueryService } from '@/application/ports';
    import type { ISeatCommandService } from '@/application/ports';
    import type { ISeatPricingService } from '@/application/ports';

    const queryClient = new QueryClient({
      defaultOptions: {
        queries: { staleTime: 5 * 60 * 1000, retry: 1 },
      },
    });

    // Stub adapters — replaced by real Cloud2 adapters in Phase 3
    const stubQueryService: ISeatPlanQueryService = {
      getSeatPlan: async () => { throw new Error('Not implemented'); },
      getSeatOccupancy: async () => { throw new Error('Not implemented'); },
    };

    const stubCommandService: ISeatCommandService = {
      assignSeat: async () => { throw new Error('Not implemented'); },
      reseatPassenger: async () => { throw new Error('Not implemented'); },
      blockSeat: async () => { throw new Error('Not implemented'); },
      unblockSeat: async () => { throw new Error('Not implemented'); },
    };

    const stubPricingService: ISeatPricingService = {
      getSeatPricing: async () => { throw new Error('Not implemented'); },
    };

    // Single DIP wiring point — ONLY place where adapters are instantiated
    export const services = {
      seatQuery: stubQueryService,
      seatCommand: stubCommandService,
      seatPricing: stubPricingService,
    } as const;

    interface Props { children: ReactNode; }

    export function DependencyProvider({ children }: Props) {
      return (
        <QueryClientProvider client={queryClient}>
          {children}
          {import.meta.env.DEV && <ReactQueryDevtools initialIsOpen={false} />}
        </QueryClientProvider>
      );
    }
    ```

    Update `src/main.tsx` to wrap App with DependencyProvider:
    ```tsx
    import { StrictMode } from 'react';
    import { createRoot } from 'react-dom/client';
    import { DependencyProvider } from '@/infrastructure/DependencyProvider';
    import App from './App';

    createRoot(document.getElementById('root')!).render(
      <StrictMode>
        <DependencyProvider>
          <App />
        </DependencyProvider>
      </StrictMode>
    );
    ```
