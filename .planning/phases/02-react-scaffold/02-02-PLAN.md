<task type="auto">
  <name>Task 2: Add `@` path alias and clean boilerplate</name>
  <complexity>simple</complexity>
  <files>vite.config.ts, tsconfig.json, src/App.tsx, src/main.tsx</files>
  <action>
    Run: `npm install -D @types/node`

    Update vite.config.ts to add resolve.alias:
    ```ts
    import path from 'path';
    // inside defineConfig:
    resolve: {
      alias: { '@': path.resolve(__dirname, './src') },
    },
    ```

    Update tsconfig.json compilerOptions to add:
    ```json
    "baseUrl": ".",
    "paths": { "@/*": ["src/*"] }
    ```

    Remove boilerplate:
    - Delete `src/App.css` and `src/assets/` directory
    - Replace `src/App.tsx` with:
      ```tsx
      export default function App() {
        return <div>poc_dcs_fe</div>;
      }
      ```
    - Ensure `src/main.tsx` uses React 18 createRoot:
      ```tsx
      import { StrictMode } from 'react';
      import { createRoot } from 'react-dom/client';
      import App from './App';

      createRoot(document.getElementById('root')!).render(
        <StrictMode>
          <App />
        </StrictMode>
      );
      ```
    - Remove the `import './index.css'` line if present in main.tsx.
      Keep `src/index.css` only if it exists and is referenced — otherwise delete it.
  </action>
  <verify>npm run lint && npm run build 2>&1 | tail -10</verify>
  <done>`npm run lint` and `npm run build` both exit 0. `@/` alias resolves correctly.</done>
</task>
